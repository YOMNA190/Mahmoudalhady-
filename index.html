# sample: conversion deduplication logic
def deduplicate_leads(leads, lookback_hours=24):
    seen = set()
    unique = []
    for lead in leads:
        uid = f"{lead['email']}_{lead['campaign_id']}"
        if uid not in seen:
            seen.add(uid)
            unique.append(lead)
    return unique
